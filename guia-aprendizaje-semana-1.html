grados = []

# Ciclo para registrar estudiantes
while True:
    print(f"\n--- REGISTRO #{contador + 1} ---")
    nombre = input("Nombre del estudiante: ")
    grado = input("Grado (ej: 5to Primaria): ")
    promedio = float(input("Promedio: "))
    
    # Agregar a las listas
    nombres.append(nombre)
    grados.append(grado)
    promedios.append(promedio)
    contador += 1
    
    # Preguntar si desea continuar
    continuar = input("\n¬øRegistrar otro estudiante? (s/n): ").lower()
    if continuar != 's':
        break

# Calcular estad√≠sticas
if contador > 0:
    promedio_grupal = sum(promedios) / len(promedios)
    promedio_mas_alto = max(promedios)
    promedio_mas_bajo = min(promedios)
    
    # Encontrar mejor y peor estudiante
    indice_mejor = promedios.index(promedio_mas_alto)
    indice_menor = promedios.index(promedio_mas_bajo)
    
    mejor_estudiante = nombres[indice_mejor]
    estudiante_bajo = nombres[indice_menor]
    
    # Contar aprobados y reprobados
    aprobados = sum(1 for p in promedios if p >= 60)
    reprobados = contador - aprobados
    
    # Contar por categor√≠a
    sobresalientes = sum(1 for p in promedios if p >= 90)
    muy_buenos = sum(1 for p in promedios if 80 <= p < 90)
    buenos = sum(1 for p in promedios if 70 <= p < 80)
    regulares = sum(1 for p in promedios if 60 <= p < 70)
    insuficientes = sum(1 for p in promedios if p < 60)
    
    # Mostrar resultados
    print("\n" + "=" * 70)
    print("           LISTADO COMPLETO DE ESTUDIANTES")
    print("=" * 70)
    
    for i in range(contador):
        estado = "‚úì APROBADO" if promedios[i] >= 60 else "‚úó REPROBADO"
        print(f"{i+1}. {nombres[i]:<30} {grados[i]:<15} {promedios[i]:>6.2f} {estado}")
    
    print("\n" + "=" * 70)
    print("           ESTAD√çSTICAS DEL GRUPO")
    print("=" * 70)
    print(f"Total de estudiantes registrados: {contador}")
    print(f"Promedio general del grupo: {promedio_grupal:.2f}")
    print(f"\nüìä Distribuci√≥n por categor√≠a:")
    print(f"  üåü Sobresaliente (90-100): {sobresalientes}")
    print(f"  üòä Muy Bueno (80-89): {muy_buenos}")
    print(f"  üëç Bueno (70-79): {buenos}")
    print(f"  ‚ö†Ô∏è  Regular (60-69): {regulares}")

    print(f"‚úó Reprobados: {reprobados}")
