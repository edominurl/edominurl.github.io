# demo_sqlite.py - Migración de JSON a SQLite import sqlite3 import json def crear_base_datos(): """Crea la tabla de estudiantes en SQLite""" conexion = sqlite3.connect('escuela.db') cursor = conexion.cursor() cursor.execute(''' CREATE TABLE IF NOT EXISTS estudiantes ( id INTEGER PRIMARY KEY AUTOINCREMENT, nombre TEXT NOT NULL, grado TEXT, promedio REAL, ausentismo INTEGER, fecha_registro DATE, estado TEXT ) ''') conexion.commit() conexion.close() print("✓ Base de datos creada exitosamente") def migrar_json_a_sqlite(): """Migra datos desde JSON a SQLite""" # Leer datos de JSON with open('estudiantes.json', 'r', encoding='utf-8') as f: estudiantes = json.load(f) # Conectar a SQLite conexion = sqlite3.connect('escuela.db') cursor = conexion.cursor() # Insertar cada estudiante for est in estudiantes: # Determinar estado según promedio if est['promedio'] < 60: estado = "En Riesgo" elif est['promedio'] < 75: estado = "Alerta" else: estado = "Normal" cursor.execute(''' INSERT INTO estudiantes (nombre, grado, promedio, ausentismo, fecha_registro, estado) VALUES (?, ?, ?, ?, ?, ?) ''', (est['nombre'], est['grado'], est['promedio'], est.get('ausentismo', 0), est.get('fecha_registro', '2025-01-01'), estado)) conexion.commit() print(f"✓ {len(estudiantes)} estudiantes migrados exitosamente") # Verificar cursor.execute('SELECT COUNT(*) FROM estudiantes') total = cursor.fetchone()[0] print(f"Total en base de datos: {total}") conexion.close() # Ejecutar migración if __name__ == "__main__": crear_base_datos() migrar_json_a_sqlite()

Operaciones CRUD Completas y Ejercicio 3

# crud_estudiantes.py - Funciones CRUD para SQLite import sqlite3 class EstudiantesDB: def __init__(self, nombre_db='escuela.db'): self.nombre_db = nombre_db def agregar_estudiante(self, nombre, grado, promedio, ausentismo, municipio): """CREATE - Agrega un nuevo estudiante""" conexion = sqlite3.connect(self.nombre_db) cursor = conexion.cursor() # Calcular nivel de riesgo if promedio < 60 or ausentismo > 10: nivel_riesgo = "Alto" elif promedio < 75 or ausentismo > 5: nivel_riesgo = "Medio" else: nivel_riesgo = "Bajo" cursor.execute(''' INSERT INTO estudiantes (nombre, grado, promedio, ausentismo, municipio, nivel_riesgo, fecha_registro) VALUES (?, ?, ?, ?, ?, ?, DATE('now')) ''', (nombre, grado, promedio, ausentismo, municipio, nivel_riesgo)) conexion.commit() estudiante_id = cursor.lastrowid conexion.close() return estudiante_id def obtener_todos(self): """READ - Obtiene todos los estudiantes""" conexion = sqlite3.connect(self.nombre_db) cursor = conexion.cursor() cursor.execute('SELECT * FROM estudiantes ORDER BY id') estudiantes = cursor.fetchall() conexion.close() return estudiantes def buscar_por_criterio(self, campo, valor): """READ - Busca estudiantes por un criterio específico""" conexion = sqlite3.connect(self.nombre_db) cursor = conexion.cursor() query = f'SELECT * FROM estudiantes WHERE {campo} LIKE ?' cursor.execute(query, (f'%{valor}%',)) resultados = cursor.fetchall() conexion.close() return resultados def obtener_estadisticas(self): """READ - Obtiene estadísticas generales""" conexion = sqlite3.connect(self.nombre_db) cursor = conexion.cursor() # Total de estudiantes cursor.execute('SELECT COUNT(*) FROM estudiantes') total = cursor.fetchone()[0] # Promedio general cursor.execute('SELECT AVG(promedio) FROM estudiantes') promedio_general = cursor.fetchone()[0] # Por nivel de riesgo cursor.execute('SELECT nivel_riesgo, COUNT(*) FROM estudiantes GROUP BY nivel_riesgo') por_riesgo = cursor.fetchall() conexion.close() return { 'total': total, 'promedio_general': round(promedio_general, 2), 'por_riesgo': dict(por_riesgo) } # Ejemplo de uso if __name__ == "__main__": db = EstudiantesDB() # Agregar estudiante nuevo_id = db.agregar_estudiante("Ana Martínez", "5to Primaria", 88.5, 2, "Quetzaltenango") print(f"Estudiante agregado con ID: {nuevo_id}") # Obtener todos todos = db.obtener_todos() print(f"\nTotal de estudiantes: {len(todos)}") # Estadísticas stats = db.obtener_estadisticas() print(f"\nEstadísticas: {stats}")

# app.py - Flask con SQLite y Templates from flask import Flask, render_template import sqlite3 app = Flask(__name__) def obtener_estudiantes(): """Obtiene todos los estudiantes de la BD""" conexion = sqlite3.connect('escuela.db') cursor = conexion.cursor() cursor.execute('SELECT * FROM estudiantes ORDER BY promedio DESC') estudiantes = cursor.fetchall() conexion.close() return estudiantes def obtener_estadisticas(): """Calcula estadísticas generales""" conexion = sqlite3.connect('escuela.db') cursor = conexion.cursor() cursor.execute('SELECT COUNT(*) FROM estudiantes') total = cursor.fetchone()[0] cursor.execute('SELECT AVG(promedio) FROM estudiantes') promedio_general = cursor.fetchone()[0] cursor.execute('SELECT COUNT(*) FROM estudiantes WHERE nivel_riesgo = "Alto"') en_riesgo_alto = cursor.fetchone()[0] conexion.close() return { 'total': total, 'promedio': round(promedio_general, 2), 'riesgo_alto': en_riesgo_alto } @app.route('/') def inicio(): estudiantes = obtener_estudiantes() stats = obtener_estadisticas() return render_template('inicio.html', estudiantes=estudiantes, estadisticas=stats) @app.route('/riesgo') def estudiantes_riesgo(): conexion = sqlite3.connect('escuela.db') cursor = conexion.cursor() cursor.execute('SELECT * FROM estudiantes WHERE nivel_riesgo IN ("Alto", "Medio") ORDER BY promedio') estudiantes = cursor.fetchall() conexion.close() return render_template('riesgo.html', estudiantes=estudiantes) if __name__ == '__main__': app.run(debug=True)

Notas Pedagógicas para el Docente

Estrategias de Enseñanza:

  • Flipped Learning: Los estudiantes ya estudiaron la guía autónoma (10 horas). La clase es 100% práctica.
  • Aprendizaje Basado en Equipos: Todos los ejercicios se resuelven en equipos de 3-4 personas.
  • Breakout Rooms: Usar salas de Teams. El docente rota cada 10-15 minutos entre salas.
  • Live Coding: Cada demostración debe hacerse en vivo, cometiendo errores intencionales y corrigiéndolos.
  • Retroalimentación Inmediata: Evaluar ejercicios mientras los estudiantes trabajan, no al final.
  • Scaffolding: Los ejercicios aumentan progresivamente en complejidad (JSON → SQLite → CRUD → Templates → Proyecto).

Gestión del Tiempo:

  • Usar cronómetro visible: Mostrar el tiempo restante en pantalla para cada ejercicio.
  • Avisar 10 min antes: "Quedan 10 minutos, enfóquense en el mínimo viable".
  • Flexibilidad: Si un equipo termina antes, asignar desafío extra (+2 puntos bonus).
  • No extender mucho: Es mejor que terminen ejercicios más simples completos que complejos incompletos.

Puntos Críticos:

  • Ejercicio 2 (SQLite): Muchos tendrán errores con rutas de archivos .db. Tener solución preparada.
  • Ejercicio 3 (CRUD): El READ con sqlite3 devuelve tuplas, no diccionarios. Aclarar esto.
  • Ejercicio 4 (Templates): La sintaxis de Jinja2 {% %} vs {{ }} confunde. Hacer énfasis.
  • Proyecto Fase 2: Algunos equipos no terminarán. Asegurar que tengan lo mínimo viable para continuar en casa.

Manejo de Equipos:

  • Roles rotativos: En cada ejercicio, cambiar quién comparte pantalla y quién escribe código.
  • Equipo rezagado: Si un equipo va muy atrasado, asignar a un estudiante avanzado como "tutor temporal".
  • Equipo muy avanzado: Darles desafíos extra (agregar gráficos, exportar a CSV, etc.).
  • Conflictos: Si hay tensión en un equipo, intervenir rápido y redistribuir tareas.

Motivación y Clima:

  • Celebrar logros: Cuando un equipo completa un ejercicio, felicitarlos públicamente.
  • Normalizar errores: "Los errores son oportunidades de aprendizaje. Yo también los cometo constantemente".
  • Conectar con lo real: "Este código que están escribiendo es lo que usan empresas como [ejemplo guatemalteco]".
  • Energía alta: Mantener un tono entusiasta, especialmente después de los recesos.
